【m_userテーブルを生成するSQL文】
CREATE TABLE m_user (
	id INT AUTO_INCREMENT PRIMARY KEY,
	user_id VARCHAR(10),
	user_mail VARCHAR(100) NOT NULL,
	user_pw VARCHAR(20) NOT NULL,
	user_name VARCHAR(20) NOT NULL,
	user_pf VARCHAR(160),
	user_img VARCHAR(50),
	user_hd VARCHAR(50)
);
UPDATE M_USER SET  user_id =(SELECT concat('u',right(concat('0000',CAST( (SELECT id FROM M_USER where user_id is null) AS CHAR)),4) )from M_USER where user_id is null ) where user_id is null;

【t_reviewテーブルを生成するSQL文】
CREATE TABLE t_review (
	id INT AUTO_INCREMENT PRIMARY KEY,
	review_id VARCHAR(10),
	video_id VARCHAR(10) NOT NULL,
	user_id VARCHAR(10) NOT NULL,
	review_contents VARCHAR(1000) NOT NULL,
	genre_id VARCHAR(10) NOT NULL,
	feelcat_name1 VARCHAR(10) NOT NULL,
	feelcat_name2 VARCHAR(10),
	star VARCHAR(1) NOT NULL,
	review_date timestamp DEFAULT CURRENT_TIMESTAMP
);
UPDATE T_REVIEW SET  review_id =(SELECT concat('u',right(concat('0000',CAST( (SELECT id FROM T_REVIEW where review_id is null) AS CHAR)),4) )from T_REVIEW where review_id is null ) where review_id is null;
